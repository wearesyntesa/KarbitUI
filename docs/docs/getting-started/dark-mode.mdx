# Dark Mode

KarbitUI supports dark mode via Tailwind CSS's class-based dark mode strategy. All components include `dark:` variant classes that activate when a `.dark` class is present on an ancestor element.

## How it works

1. Add `class="dark"` to your `<html>` element
2. All KarbitUI components automatically switch to dark mode colors
3. Remove the class to switch back to light mode

```html
<!-- Light mode (default) -->
<html lang="en">

<!-- Dark mode -->
<html lang="en" class="dark">
```

## Manual toggle

```ts
function toggleDarkMode(): void {
  document.documentElement.classList.toggle("dark");
}
```

## System preference detection

```ts
function initDarkMode(): void {
  const prefersDark = window.matchMedia("(prefers-color-scheme: dark)");

  function apply(dark: boolean): void {
    document.documentElement.classList.toggle("dark", dark);
  }

  apply(prefersDark.matches);
  prefersDark.addEventListener("change", (e) => { apply(e.matches); });
}

initDarkMode();
```

## Persistent preference

```ts
type Theme = "light" | "dark" | "system";

function setTheme(theme: Theme): void {
  localStorage.setItem("theme", theme);
  applyTheme(theme);
}

function applyTheme(theme: Theme): void {
  const isDark =
    theme === "dark" ||
    (theme === "system" &&
      window.matchMedia("(prefers-color-scheme: dark)").matches);

  document.documentElement.classList.toggle("dark", isDark);
}

// On page load
const stored = localStorage.getItem("theme") as Theme | null;
applyTheme(stored ?? "system");
```

## React example

```tsx
import { useState, useEffect } from "react";

type Theme = "light" | "dark" | "system";

function useTheme() {
  const [theme, setTheme] = useState<Theme>(() => {
    return (localStorage.getItem("theme") as Theme) ?? "system";
  });

  useEffect(() => {
    const isDark =
      theme === "dark" ||
      (theme === "system" &&
        window.matchMedia("(prefers-color-scheme: dark)").matches);

    document.documentElement.classList.toggle("dark", isDark);
    localStorage.setItem("theme", theme);
  }, [theme]);

  return { theme, setTheme };
}
```

## Tailwind v4 configuration

The KarbitUI theme CSS already configures dark mode via class strategy:

```css
/* Already included in @wearesyntesa/karbit-ui/theme.css */
@custom-variant dark (&:where(.dark, .dark *));
```

This means `dark:` utilities only activate when a `.dark` class is present on an ancestor element. No additional Tailwind configuration is needed.

## Color tokens

| Token | Light | Dark |
|---|---|---|
| Surface | `#ffffff` | `#18181b` (zinc-900) |
| Surface muted | `#f8f9fa` | `#27272a` (zinc-800) |
| Text primary | `#0f172a` (slate-900) | `#fafafa` (zinc-50) |
| Text secondary | `#64748b` (slate-500) | `#a1a1aa` (zinc-400) |
| Text muted | `#94a3b8` (slate-400) | `#71717a` (zinc-500) |
| Border | `#e5e7eb` (gray-200) | `#3f3f46` (zinc-700) |
| Accent | `#3b82f6` (blue-500) | `#60a5fa` (blue-400) |
